Register CPTs in a separate file. Use consistent naming and structure.

Example: CPT Registration 
<?php
// inc/post-types.php

function register_project_post_type() {
  $labels = [
    'name' => 'Projects',
    'singular_name' => 'Project',
    'menu_name' => 'Projects',
    'add_new' => 'Add New',
    'add_new_item' => 'Add New Project',
    'edit_item' => 'Edit Project',
    'new_item' => 'New Project',
    'view_item' => 'View Project',
    'search_items' => 'Search Projects',
    'not_found' => 'No projects found',
    'not_found_in_trash' => 'No projects found in Trash',
  ];
  
  $args = [
    'labels' => $labels,
    'public' => true,
    'publicly_queryable' => true,
    'show_ui' => true,
    'show_in_menu' => true,
    'query_var' => true,
    'rewrite' => ['slug' => 'projects'],
    'capability_type' => 'post',
    'has_archive' => true,
    'hierarchical' => false,
    'menu_position' => 20,
    'menu_icon' => 'dashicons-portfolio',
    'supports' => ['title', 'editor', 'thumbnail', 'excerpt'],
    'show_in_rest' => true, // Enable Gutenberg
  ];
  
  register_post_type('project', $args);
}
add_action('init', 'register_project_post_type');

Use ACF JSON for version control. Always sanitize and escape ACF field values.

Example: ACF Field Usage 
<?php
// Get ACF field values
$title = get_field('title');
$content = get_field('content');
$image = get_field('image');
$link = get_field('link');
$repeater = get_field('items');

// Always escape output
<h2><?php echo esc_html($title); ?></h2>
<div><?php echo wp_kses_post($content); ?></div>

// Image field
if ($image) {
  $image_url = esc_url($image['url']);
  $image_alt = esc_attr($image['alt']);
  ?>
  <img src="<?php echo $image_url; ?>" alt="<?php echo $image_alt; ?>">
  <?php
}

// Link field
if ($link) {
  $link_url = esc_url($link['url']);
  $link_title = esc_html($link['title']);
  $link_target = $link['target'] ? ' target="_blank" rel="noopener noreferrer"' : '';
  ?>
  <a href="<?php echo $link_url; ?>"<?php echo $link_target; ?>>
    <?php echo $link_title; ?>
  </a>
  <?php
}

// Repeater field
if ($repeater) {
  foreach ($repeater as $item) {
    $item_title = esc_html($item['title']);
    $item_content = wp_kses_post($item['content']);
    ?>
    <div class="item">
      <h3><?php echo $item_title; ?></h3>
      <div><?php echo $item_content; ?></div>
    </div>
    <?php
  }
}

Use template parts for reusable components. Pass data via arguments.

Example: Template Part Usage 
<?php
// template-parts/components/card.php
$args = wp_parse_args($args, [
  'title' => '',
  'content' => '',
  'image' => null,
  'link' => null,
  'class' => '',
]);

$title = esc_html($args['title']);
$content = wp_kses_post($args['content']);
$image = $args['image'];
$link = $args['link'];
$class = esc_attr($args['class']);
?>

<article class="card <?php echo $class; ?>">
  <?php if ($image) : ?>
    <div class="card__image">
      <img src="<?php echo esc_url($image['url']); ?>" 
           alt="<?php echo esc_attr($image['alt']); ?>">
    </div>
  <?php endif; ?>
  
  <div class="card__content">
    <h3 class="card__title"><?php echo $title; ?></h3>
    <div class="card__text"><?php echo $content; ?></div>
    
    <?php if ($link) : ?>
      <a href="<?php echo esc_url($link['url']); ?>" 
         class="card__link">
        <?php echo esc_html($link['title']); ?>
      </a>
    <?php endif; ?>
  </div>
</article>

<?php
// Usage in template
$card_args = [
  'title' => get_field('title'),
  'content' => get_field('content'),
  'image' => get_field('image'),
  'link' => get_field('link'),
  'class' => 'card--featured',
];
get_template_part('template-parts/components', 'card', $card_args);

Use actions and filters properly. Remove hooks when needed.

Example: Hooks Usage 
<?php
// Actions
add_action('wp_enqueue_scripts', 'enqueue_theme_scripts');
add_action('after_setup_theme', 'theme_setup');
add_action('wp_footer', 'add_custom_script');

function enqueue_theme_scripts() {
  wp_enqueue_style('theme-style', get_stylesheet_uri());
  wp_enqueue_script('theme-script', get_template_directory_uri() . '/assets/js/main.js', [], '1.0.0', true);
}

// Filters
add_filter('excerpt_length', 'custom_excerpt_length', 999);
add_filter('the_content', 'add_custom_content');

function custom_excerpt_length($length) {
  return 20;
}

function add_custom_content($content) {
  if (is_single()) {
    $content .= '<p>Custom content</p>';
  }
  return $content;
}

// Remove hooks
remove_action('wp_head', 'wp_generator');
remove_filter('the_content', 'wpautop');
